車種リスト・スコア

# ライブラリ読み込み
library(readxl)
library(dplyr)

Attaching package: 'dplyr'
The following objects are masked from 'package:stats':

    filter, lag
The following objects are masked from 'package:base':

    intersect, setdiff, setequal, union
library(DT)

# Excelファイルの読み込み
file_path <- "バイク年式;燃費 .xlsx"
sheet_names <- excel_sheets(file_path)

# 各シートを読み込み、メーカー列を追加
all_sheets <- lapply(sheet_names, function(sheet) {
  df <- read_excel(file_path, sheet = sheet)
  df$メーカー <- sheet
  df
})
all_bikes <- bind_rows(all_sheets)

# 列名を標準化（列の順番に応じて調整が必要な場合あり）
all_bikes <- all_bikes %>%
  rename(
    車種 = 1,
    年式 = 2,
    燃費 = 3,
    タンク容量 = 4,
    航続距離 = 5,
    排気量 = 6
  ) %>%
  mutate(
    年式 = as.numeric(年式),
    燃費 = as.numeric(燃費),
    タンク容量 = as.numeric(タンク容量),
    航続距離 = as.numeric(航続距離),
    排気量 = as.numeric(排気量)
  )

# 各項目をスコア化（年式、燃費、タンク容量、航続距離、排気量＝7段階）
all_bikes <- all_bikes %>%
  mutate(
    年式スコア = case_when(
      年式 >= 2020 ~ 5,
      年式 >= 2015 ~ 4,
      年式 >= 2010 ~ 3,
      年式 >= 2000 ~ 2,
      年式 >= 1990 ~ 1,
      TRUE ~ 0
    ),
    燃費スコア = case_when(
      燃費 >= 40 ~ 5,
      燃費 >= 35 ~ 4,
      燃費 >= 30 ~ 3,
      燃費 >= 25 ~ 2,
      燃費 >= 20 ~ 1,
      TRUE ~ 0
    ),
    タンクスコア = case_when(
      タンク容量 >= 20 ~ 5,
      タンク容量 >= 17 ~ 4,
      タンク容量 >= 14 ~ 3,
      タンク容量 >= 10 ~ 2,
      タンク容量 >= 5 ~ 1,
      TRUE ~ 0
    ),
    航続スコア = case_when(
      航続距離 >= 500 ~ 5,
      航続距離 >= 400 ~ 4,
      航続距離 >= 300 ~ 3,
      航続距離 >= 200 ~ 2,
      航続距離 >= 100 ~ 1,
      TRUE ~ 0
    ),
    排気量スコア = case_when(
      排気量 <= 50 ~ 1,
      排気量 <= 125 ~ 2,
      排気量 <= 250 ~ 3,
      排気量 <= 400 ~ 4,
      排気量 <= 750 ~ 5,
      排気量 <= 1000 ~ 6,
      排気量 > 1000 ~ 7,
      TRUE ~ NA_real_
    ),
    合計スコア = 年式スコア + 燃費スコア + タンクスコア + 航続スコア + 排気量スコア
  )

# 表形式で表示（インタラクティブ）
datatable(
  all_bikes %>%
    select(メーカー, 車種, 年式, 燃費, タンク容量, 航続距離, 排気量,
           年式スコア, 燃費スコア, タンクスコア, 航続スコア, 排気量スコア, 合計スコア),
  options = list(pageLength = 15),
  caption = "バイクのスペック評価（スコア一覧）"
)

メーカー別平均燃費

# 必要なパッケージ
library(dplyr)
library(DT)

# データ処理 + 表を作成
all_bikes %>%
  group_by(メーカー) %>%
  summarise(平均燃費 = mean(燃費, na.rm = TRUE)) %>%
  arrange(desc(平均燃費)) %>%
  datatable(options = list(pageLength = 10, autoWidth = TRUE))
  • 燃費重視のユーザーにとっては、日本メーカーのバイクが有力な選択肢となります。
  • 外車は燃費よりも**走行性能・デザイン・エンジン特性（大排気量や高出力）**が重視されている可能性があります

排気量別平均燃費

library(dplyr)
library(DT)

# 排気量帯を追加
all_bikes <- all_bikes %>%
  mutate(
    排気量帯 = case_when(
      排気量 <= 50 ~ "〜50cc",
      排気量 <= 125 ~ "51〜125cc",
      排気量 <= 250 ~ "126〜250cc",
      排気量 <= 400 ~ "251〜400cc",
      排気量 <= 750 ~ "401〜750cc",
      排気量 <= 1000 ~ "751〜1000cc",
      排気量 > 1000 ~ "1001cc〜",
      TRUE ~ "不明"
    )
  )

# 平均燃費を排気量帯ごとに計算し、インタラクティブ表に
all_bikes %>%
  group_by(排気量帯) %>%
  summarise(平均燃費 = mean(燃費, na.rm = TRUE)) %>%
  arrange(排気量帯) %>%
  datatable(
    caption = "排気量帯別の平均燃費（km/L）",
    options = list(pageLength = 10, autoWidth = TRUE)
  )

  • 排気量が小さいほど燃費が良い

  • 特に「50cc以下」「51〜125cc」は50km/Lを超える高燃費。

  • 小排気量車（スクーターや原付）は燃費性能が非常に高いことが分かります。

  • 排気量が大きくなるにつれて燃費は悪化

  • 「751〜1000cc」「1001cc以上」では20km/L未満と低め。

  • 排気量が大きいバイクはパワーとトルク重視の設計で、燃費は犠牲になっている傾向。

  • 126〜400ccはバランス型

  • この帯域（126〜400cc）は燃費とパワーの中間的なポジション

  • 初心者から中級者向けに人気がある理由の一つと考えられ

メーカー別・排気量別平均燃費

library(dplyr)
library(tidyr)
library(DT)

all_bikes <- all_bikes %>%
  mutate(
    年式帯 = case_when(
      年式 >= 2020 ~ "2020年代",
      年式 >= 2010 ~ "2010年代",
      年式 >= 2000 ~ "2000年代",
      年式 >= 1990 ~ "1990年代",
      年式 >= 1980 ~ "1980年代",
      TRUE ~ "その他"
    )
  )

table1 <- all_bikes %>%
  group_by(メーカー, 排気量帯) %>%
  summarise(平均燃費 = round(mean(燃費, na.rm = TRUE), 1), .groups = "drop") %>%
  pivot_wider(names_from = 排気量帯, values_from = 平均燃費)

datatable(table1, caption = "📊 メーカー × 排気量帯ごとの平均燃費", options = list(pageLength = 20))

🔍 全体的な読み取りポイント

✅ 排気量が小さいほど燃費が良い傾向は全メーカー共通

例：

ホンダ：

  • ～50cc：64.5 km/L

  • 51〜125cc：62.7 km/L

  • 1001cc〜：17.7 km/L

  • スズキ：

  • ～50cc：75.9 km/L（表中最高値）

  • 1001cc〜：20 km/L

    ✅ ヤマハは全体的に燃費が優秀

  • どの排気量帯でも比較的高め（例：126〜250ccで35.4km/L、400cc以上でも27.2km/L）

✅ **欧州メーカー（BMW、KTM、トライアンフ）は大排気量で苦戦**

  • KTM（1001cc〜）：17.0 km/L

  • トライアンフ（1001cc〜）：20.3 km/L

  • BMW（1001cc〜）：21.1 km/L

  • 一方でBMWの126〜250ccは比較的良好（30.0 km/L）

    🏆 特筆すべきポイント

特徴 内容
✅ 燃費最良（全体） スズキ ～50cc：75.9 km/L
✅ 小排気量で特に優秀なメーカー ホンダ・スズキ・ヤマハ（全て50km/L超え）
❌ 燃費が低い傾向のメーカー KTM（全体的に20〜30km/L以下が多い）
✅ 250〜400cc帯で燃費が高い ヤマハ（33.2 km/L）
❌ 極端な燃費のばらつき トライアンフ（400〜750ccは良いが、50cc以下や高排気量は不明／低め）

🧠 考察

  • ホンダ・ヤマハ・スズキは全排気量帯で安定した燃費性能を見せており、特に原付や軽二輪で圧倒的な燃費性能を発揮。

  • BMW・KTM・トライアンフなどの海外メーカーは、特に高排気量車が多く、燃費よりもパフォーマンス志向。

  • アプリリアは中排気量帯で控えめな性能だが、50cc以下では42.3 km/Lと比較的良い数値。

  • 📌 まとめ

  • 燃費重視なら日本メーカー（特にスズキ・ホンダ・ヤマハ）の小〜中排気量バイクが有力候補。

**趣味・ツーリング用途の欧州車は燃費よりもエンジン特性・ブランド性**を重視する傾向。

📊 全体傾向

1. 排気量が小さいほど燃費が良い（全メーカー共通）

  • これはエンジンの物理的特性・車体重量に基づく自然な傾向です。
例：

  • ホンダ（～50cc）：64.5 km/L

  • ヤマハ（～50cc）：65.6 km/L

  • スズキ（～50cc）：75.9 km/L（表中最高）

🏆 メーカー別の特徴

✅ ホンダ

  • ～125cc帯で非常に優秀（51～125cc：62.7km/L）

  • 大排気量では控えめ（1001cc〜：17.7km/L）

  • 全体的に燃費の安定感が高い

✅ ヤマハ

  • 全排気量でバランスよく燃費が良い

  • 126〜250cc：35.4km/L（表中最高）

  • 401〜750cc：27.2km/L（他社より高い）

  • 総合的に最も燃費優秀なメーカー

✅ スズキ

  • ～50ccで75.9km/Lと最高記録

  • 小〜中排気量で堅実な燃費（例：51〜125cc 45.4km/L）

    KTM

  • 全体的に燃費は低め

    1001cc〜：17.0km/L（最低）

    401〜750cc：データなし

    高出力志向のブランドらしい傾向

⚠️ BMW

  • 小〜中排気量では意外に健闘（126〜250cc：30km/L）

  • 高排気量は平凡（1001cc〜：21.1km/L）

    アプリリア・トライアンフ

  • データが一部欠けている（“不明”列あり）

アプリリア：50ccは42.3km/L → 中程度

トライアンフ：全体的に燃費はやや低め

📌 まとめ：この表から読み取れる主なこと

観点 結論
燃費最強排気量帯 ～50cc（全メーカーで高燃費）
燃費最強メーカー ヤマハ（全体的にバランスが良く、特に中排気量で優秀）
高排気量で健闘 BMW（1001cc以上で21.1km/Lは高め）
低燃費メーカー KTM（すべての帯域で20〜30km/L未満）
特筆すべき記録 スズキ ～50ccで75.9km/L → 全体で最高値
排気量と燃費の関係 排気量が上がるほど燃費が落ちる傾向は明確
